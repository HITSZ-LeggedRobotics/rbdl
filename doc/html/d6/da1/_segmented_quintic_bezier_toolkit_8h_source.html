599  /*
600  This function numerically integrates the Bezier curve y(x).
601 
602  @param vX Values of x to evaluate the integral of y(x)
603  @param ic0 The initial value of the integral
604  @param intAcc Accuracy of the integrated solution
605  @param uTol Tolerance on the calculation of the intermediate u term
606  @param uMaxIter Maximum number of iterations allowed for u to reach its
607  desired tolerance.
608  @param mX The 6xn matrix of Bezier control points for x(u)
609  @param mY The 6xn matrix of Bezier control points for y(u)
610 
611  @param flag_intLeftToRight Setting this flag to true will evaluate the
612  integral from the left most point to the right most
613  point. Setting this flag to false will cause the
614  integral to be evaluated from right to left.
615  @param name Name of caller.
616  @return RigidBodyDynamics::Math::MatrixNd Col 0: X vector, Col 1: int(y(x))
617 
618 
619  This function numerically integrates the Bezier curve y(x), when really
620  both x and y are specified in terms of u. Evaluate the integral at the
621  locations specified in vX and return the result.
622 
623  <B>Computational Costs</B>
624 
625  This the expense of this function depends on the number of points in
626  vX, the points for which the integral y(x) must be calculated. The
627  integral is evaluated using a Runge Kutta 45 integrator, and so each
628  point requires 6 function evaluations.
629  (http://en.wikipedia.org/wiki/Dormand%E2%80%93Prince_method)
630 
631  The cost of evaluating 1 Bezier curve y(x) scales with the number
632  of points in xVal:
633  \verbatim
634  ~1740 flops per point
635  \endverbatim
636 
637  The example below is quite involved, but just so it can show you an
638  example of how to create the array of Spline objects that approximate
639  the function u(x). Although the example has been created for only 1
640  Bezier curve set, simply changing the size and entries of the matricies
641  _mX and _mY will allow multiple sets to be integrated.
642 
643 
644  <B>Example:</B>
645  @code
646  //Integrator and u tolerance settings
647  double INTTOL = 1e-12;
648  double UTOL = 1e-14;
649  int MAXITER= 10;
650 
651  //Make up a Bezier curve - these happen to be the control points
652  //for a tendon curve
653  RigidBodyDynamics::Math::MatrixNd _mX(6,1), _mY(6,1);
654  _mX(0)= 1;
655  _mX(1)= 1.01164;
656  _mX(2)= 1.01164;
657  _mX(3)= 1.02364;
658  _mX(4)= 1.02364;
659  _mX(5)= 1.04;
660 
661  _mY(0) = 0;
662  _mY(1) = 3.10862e-16;
663  _mY(2) = 3.10862e-16;
664  _mY(3) = 0.3;
665  _mY(4) = 0.3;
666  _mY(5) = 1;
667 
668  _numBezierSections = 1;
669  bool _intx0x1 = true; //Says we're integrating from x0 to x1
671  //Generate the set of splines that approximate u(x)
673  RigidBodyDynamics::Math::VectorNd u(NUM_SAMPLE_PTS);
674  //Used for the approximate inverse
675  RigidBodyDynamics::Math::VectorNd x(NUM_SAMPLE_PTS);
676  //Used for the approximate inverse
677 
678  //Used to generate the set of knot points of the integral of y(x)
679  RigidBodyDynamics::Math::VectorNd
680  xALL(NUM_SAMPLE_PTS*_numBezierSections-(_numBezierSections-1));
681  _arraySplineUX.resize(_numBezierSections);
682  int xidx = 0;
683 
684  for(int s=0; s < _numBezierSections; s++){
685  //Sample the local set for u and x
686  for(int i=0;i<NUM_SAMPLE_PTS;i++){
687  u(i) = ( (double)i )/( (double)(NUM_SAMPLE_PTS-1) );
688  x(i) = SegmentedQuinticBezierToolkit::
689  calcQuinticBezierCurveVal(u(i),_mX(s),_name);
690  if(_numBezierSections > 1){
691  //Skip the last point of a set that has another set of points
692  //after it. Why? The last point and the starting point of the
693  //next set are identical in value.
694  if(i<(NUM_SAMPLE_PTS-1) || s == (_numBezierSections-1)){
695  xALL(xidx) = x(i);
696  xidx++;
697  }
698  }else{
699  xALL(xidx) = x(i);
700  xidx++;
701  }
702  }
703  //Create the array of approximate inverses for u(x)
704  _arraySplineUX[s] = SimTK::SplineFitter<double>::
705  fitForSmoothingParameter(3,x,u,0).getSpline();
706  }
707 
708 
710  //Compute the integral of y(x) and spline the result
712 
713  RigidBodyDynamics::Math::VectorNd yInt = SegmentedQuinticBezierToolkit::
714  calcNumIntBezierYfcnX(xALL,0,INTTOL, UTOL, MAXITER,_mX, _mY,
715  _arraySplineUX,_name);
716 
717  if(_intx0x1==false){
718  yInt = yInt*-1;
719  yInt = yInt - yInt(yInt.nelt()-1);
720  }
721 
722  _splineYintX = SimTK::SplineFitter<double>::
723  fitForSmoothingParameter(3,xALL,yInt,0).getSpline();
724 
725  @endcode
726
