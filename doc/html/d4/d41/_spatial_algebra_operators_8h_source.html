52  Vector3d mv_lower (mv[3], mv[4], mv[5]);

54  Vector3d res_upper = Vector3d (

198  SpatialRigidBodyInertia apply (const SpatialRigidBodyInertia &rbi) {
199  return SpatialRigidBodyInertia (
200  rbi.m,
201  E * (rbi.h - rbi.m * r),
202  E *
